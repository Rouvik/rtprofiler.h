254void *bmalloc(size_t size)
255{
256 size_t *ptr = malloc(size + sizeof(size_t));
257 if (!ptr)
258 {
259 return NULL;
260 }
261
262 *ptr = size;
263 BENCH_HEAP_TOTAL += size;
264 BENCH_HEAP_CURRENT += size;
265 return ptr + 1;
266}

268size_t bGetMallocSize(void *ptr)
269{
270 if (!ptr)
271 {
272 return 0;
273 }
274
275 size_t *size = (size_t *)ptr - 1;
276 return *size;
277}

279void bfree(void *ptr)
280{
281 if (!ptr)
282 {
283 return;
284 }
285
286 size_t *alloc_size = (size_t *)ptr - 1;
287 BENCH_HEAP_CURRENT -= *alloc_size;
288 free(alloc_size);
289}
