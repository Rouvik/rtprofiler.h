279void *bmalloc(size_t size)
280{
281 size_t *ptr = malloc(size + sizeof(size_t));
282 if (!ptr)
283 {
284 return NULL;
285 }
286
287 *ptr = size;
288 BENCH_HEAP_TOTAL += size;
289 BENCH_HEAP_CURRENT += size;
290 return ptr + 1;
291}

293size_t bGetMallocSize(void *ptr)
294{
295 if (!ptr)
296 {
297 return 0;
298 }
299
300 size_t *size = (size_t *)ptr - 1;
301 return *size;
302}

304void bfree(void *ptr)
305{
306 if (!ptr)
307 {
308 return;
309 }
310
311 size_t *alloc_size = (size_t *)ptr - 1;
312 BENCH_HEAP_CURRENT -= *alloc_size;
313 free(alloc_size);
314}
